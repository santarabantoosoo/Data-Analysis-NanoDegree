def num_list(the_list):
    the_len = len(the_list)
    auto = []
    manual = []
    for i in range(the_len):
        if len(the_list[i]) == 2:
            auto.append(the_list[i][0][1])
            manual.append(the_list[i][1][1])
        if len(the_list[i]) == 1:
            auto.append(the_list[i][0][1])
            manual.append(0)
    return [auto, manual]

def fuel_counter(the_list):
    the_len = len(the_list)
    flex = np.zeros(the_len)
    regular = np.zeros(the_len)
    premium = np.zeros(the_len)
    for i in range(the_len-1):
        if len(the_list[i]) == 1:
            regular[i] = the_list[i][0][1]
            
            
        if len(the_list[i]) == 2:
            if the_list[i][0][0] == "flex-fuel (unleaded/E85)":
                flex[i] = the_list[i][0][1]
            elif the_list[i][0][0] == "regular unleaded":
                regular[i] = the_list[i][0][1]
            else:
                premium[i] = the_list[i][0][1]
                
            if the_list[i][1][0] == "flex-fuel (unleaded/E85)":
                flex[i] = the_list[i][1][1]
            elif the_list[i][1][0] == "regular unleaded":
                regular[i] = the_list[i][1][1]
            else:
                premium[i] = the_list[i][1][1]
    return [flex, regular, premium]

def make_counter(the_list):
    the_len = len(the_list)
    audi = np.zeros(the_len)
    audi[:] = np.nan
    ford = np.zeros(the_len)
    ford[:] = np.nan
    honda = np.zeros(the_len)
    honda[:] = np.nan
    toyota = np.zeros(the_len)
    toyota[:] = np.nan
    bmw = np.zeros(the_len)
    bmw[:] = np.nan

    for i in range(the_len):
        if len(the_list[i]) == 1:
            ford[i] = 10
        if len(the_list[i]) == 2:
            audi[i] = 6
            ford[i] = 4
        if len(the_list[i]) == 3:
            if the_list[i][0][0] == "Audi":
                audi[i] = the_list[i][0][1]
            if the_list[i][0][0] == "Ford":
                ford[i] = the_list[i][0][1]
            if the_list[i][0][0] == "Honda":
                honda[i] = the_list[i][0][1]
            if the_list[i][0][0] == "Toyota":
                toyota[i] = the_list[i][0][1]
            if the_list[i][0][0] == "BMW":
                bmw[i] = the_list[i][0][1]
                
            if the_list[i][1][0] == "Audi":
                audi[i] = the_list[i][1][1]
            if the_list[i][1][0] == "Ford":
                ford[i] = the_list[i][1][1]
            if the_list[i][1][0] == "Honda":
                honda[i] = the_list[i][1][1]
            if the_list[i][1][0] == "Toyota":
                toyota[i] = the_list[i][1][1]
            if the_list[i][1][0] == "BMW":
                bmw[i] = the_list[i][1][1]
